## 工具方法

```js
// 只提取汉字
GetChinese(strValue) { // eslint-disable-line
    if (strValue !== null && strValue !== '') {
        const reg = /[\u4e00-\u9fa5]/g;
        return strValue.match(reg).join('');
    }
    return '';
}
// 去掉部字符串里面所有的符号
RemoveSymbol(str) {
   return str.replace(/[^\u4e00-\u9fa5\w]/g,"")
}
```
